cover


            var msg = function(second){
                return "敌军还有" + second + "秒到达战场"
            }

            var surplus = 5;

            var cover = pop.cover({
                msg     : msg(surplus),
                onopen : function(info){
                    var timer = setInterval(function(){
                        surplus--;
                        if(surplus){
                            cover.Update({
                                msg     : msg(surplus),
                            })
                        }else{
                            clearInterval(timer);
                            cover.Hide();
                        }
                    },1000)
                },
                onclose : function(){
                    console.log("敌军到达战场");
                },
            });
